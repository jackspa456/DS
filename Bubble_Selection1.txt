#include<iostream>
using namespace std;

typedef struct student
{
    int roll_num;
    char name[20];
    float marks;
} stud;

void create(stud s[20], int n);
void display(stud s[20], int n);
void bubble_sort(stud s[20], int n);
void selection_sort(stud s[20], int n);

int main()
{
    stud s[20];
    int ch, n;

    do
    {
        cout<<"\n 1) Create Student Database ";
        cout<<"\n 2) Display Student Records ";
        cout<<"\n 3) Bubble Sort ";
        cout<<"\n 4) Selection Sort ";
        cout<<"\n 5) Exit ";
        cout<<"\n Enter Your Choice: ";
        cin>>ch;

        switch(ch)
        {
            case 1:
                cout<<"\n Enter the number of records (minimum 10): ";
                cin>>n;
                create(s, n);
                break;

            case 2:
                display(s, n);
                break;

            case 3:
                bubble_sort(s, n);
                cout<<"\n Records Sorted Using Bubble Sort!";
                break;

            case 4:
                selection_sort(s, n);
                cout<<"\n Records Sorted Using Selection Sort!";
                break;

            case 5:
                return 0;

            default:
                cout<<"\n Invalid Choice!";
        }
    } while(ch != 5);
}


void create(stud s[20], int n)
{
    for(int i=0; i<n; i++)
    {
        cout<<"\n Enter Roll Number: ";
        cin>>s[i].roll_num;
        cout<<" Enter Name: ";
        cin>>s[i].name;
        cout<<" Enter Marks: ";
        cin>>s[i].marks;
    }
}


void display(stud s[20], int n)
{
    cout<<"\n\tRoll No\t Name\t Marks";
    for(int i=0; i<n; i++)
    {
        cout<<"\n\t "<< s[i].roll_num <<"\t "<< s[i].name <<"\t "<< s[i].marks;
    }
}


void bubble_sort(stud s[20], int n)
{
    for(int i=0; i<n-1; i++)
    {
        for(int j=0; j<n-i-1; j++)
        {
            if(s[j].roll_num > s[j+1].roll_num)
            {
                stud temp = s[j];
                s[j] = s[j+1];
                s[j+1] = temp;
            }
        }
    }
}


void selection_sort(stud s[20], int n)
{
    for(int i=0; i<n-1; i++)
    {
        int min_index = i;
        for(int j=i+1; j<n; j++)
        {
            if(s[j].roll_num < s[min_index].roll_num)
                min_index = j;
        }

        if(min_index != i)
        {
            stud temp = s[i];
            s[i] = s[min_index];
            s[min_index] = temp;
        }
    }
}
