text = input("Enter text:")
pattern = input("Enter pattern:")

n = len(text)
m = len(pattern)

found = False
print("Checking for pattern in text...")
for i in range(n - m + 1):
    j = 0
    while j < m and text[i + j] == pattern[j]:
        j += 1
    if j == m:
        print("Pattern found at positions:")
        print(f"Pattern found at in index{i}")
        found = True

if not found:
    print("No Pattern Match found in the string")
