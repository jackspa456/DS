#include <iostream>
using namespace std;


struct Node {
    int data;
    Node* next;
};


bool isEmpty(Node* top) {
    return (top == NULL);
}


bool isFull() {
    Node* temp = new(nothrow) Node;  
    if (temp == NULL)
        return true;  
    delete temp;
    return false;
}


void push(Node*& top, int value) {
    if (isFull()) {
        cout << "Stack Overflow! Memory full.";
        return;
    }

    Node* newNode = new Node;
    newNode->data = value;
    newNode->next = top;
    top = newNode;
    cout << value << " pushed onto stack.";
}


void pop(Node*& top) {
    if (isEmpty(top)) {
        cout << "Stack Underflow! Nothing to pop.";
    } else {
        Node* temp = top;
        cout << "Popped element: " << temp->data;
        top = top->next;
        delete temp;
    }
}


void peek(Node* top) {
    if (isEmpty(top))
        cout << "Stack is empty.";
    else
        cout << "Top element is: " << top->data;
}


void display(Node* top) {
    if (isEmpty(top))
        cout << "Stack is empty.";
    else {
        cout << "Stack elements (top to bottom): ";
        Node* temp = top;
        while (temp != NULL) {
            cout << temp->data << " ";
            temp = temp->next;
        }
        cout << endl;
    }
}

int main() {
    Node* top = NULL;
    int choice, value;

    do {
        cout << "\n--- Stack Menu ---";
        cout << "\n1. Push";
        cout << "\n2. Pop";
        cout << "\n3. Peek";
        cout << "\n4. IsEmpty";
        cout << "\n5. IsFull";
        cout << "\n6. Display";
        cout << "\n7. Exit";
        cout << "\nEnter choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter value to push: ";
                cin >> value;
                push(top, value);
                break;

            case 2:
                pop(top);
                break;

            case 3:
                peek(top);
                break;

            case 4:
                if (isEmpty(top))
                    cout << "Stack is empty.";
                else
                    cout << "Stack is not empty.";
                break;

            case 5:
                if (isFull())
                    cout << "Stack is full.";
                else
                    cout << "Stack is not full.";
                break;

            case 6:
                display(top);
                break;

            case 7:
                cout << "Exiting program.";
                break;

            default:
                cout << "Invalid choice! Try again.";
        }
    } while (choice != 7);

    return 0;
}
